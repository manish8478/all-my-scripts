function getOrCreateFolder(parentFolder, folderName) {
  var iter = parentFolder.getFoldersByName(folderName);
  if (iter.hasNext()) return iter.next();
  return parentFolder.createFolder(folderName);
}

function uploadFileToDrive(base64Data, fileName, mimeType, folder) {
  try {
    var decoded = Utilities.base64Decode(base64Data);
    var blob    = Utilities.newBlob(decoded, mimeType, fileName);
    var file    = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return file.getUrl();
  } catch(e) {
    return "Upload error: " + e.message;
  }
}

function formatDateDMY(dateStr) {
  // Input: "14/3/2026"  →  Output: "14-03-2026"
  var parts = dateStr.split("/");
  if (parts.length === 3) {
    var d = parts[0].length === 1 ? "0" + parts[0] : parts[0];
    var m = parts[1].length === 1 ? "0" + parts[1] : parts[1];
    return d + "-" + m + "-" + parts[2];
  }
  return dateStr.replace(/\//g, "-");
}

// ── Upload single file to Drive ──
// Drive structure: Cleanness Audit (main) → "Fish Plate-14-03-2026" → image
function uploadSingleFile(fileData) {
  try {
    var mainFolder    = DriveApp.getFolderById(MAIN_FOLDER_ID);
    var dateFormatted = formatDateDMY(fileData.date);

    // Subfolder name: "Fish Plate-14-03-2026"
    var subFolderName = fileData.division + "-" + dateFormatted;
    var iter          = mainFolder.getFoldersByName(subFolderName);
    var subFolder     = iter.hasNext() ? iter.next() : mainFolder.createFolder(subFolderName);

    var url = uploadFileToDrive(fileData.base64, fileData.name, fileData.mimeType, subFolder);
    return { status: "success", qno: parseInt(fileData.qno), url: url };
  } catch(e) {
    return { status: "error", qno: fileData.qno, msg: e.message };
  }
}

// ── Save audit rows to sheet ──
// Attachment column mein Drive URL directly store hoga
function saveAuditRows(payload) {
  if (!payload || !payload.rows || payload.rows.length === 0)
    return { status: "error", msg: "No rows" };

  var rows       = payload.rows;
  var driveLinks = payload.driveLinks || {};

  var ss    = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName(MASTER_SHEET);

  if (!sheet) {
    sheet = ss.insertSheet(MASTER_SHEET);
    sheet.appendRow(MASTER_HEADERS);
    var hr = sheet.getRange(1, 1, 1, MASTER_HEADERS.length);
    hr.setBackground("#006b3f");
    hr.setFontColor("#ffffff");
    hr.setFontWeight("bold");
    hr.setHorizontalAlignment("center");
    sheet.setFrozenRows(1);
  }

  rows.forEach(function(r) {
    var qnoKey   = parseInt(r.qno);
    // Drive URL for this question row — blank if no photo attached
    var driveUrl = driveLinks[qnoKey] || driveLinks[String(qnoKey)] || "";

    sheet.appendRow([
      r.date        || "",   // Date
      r.time        || "",   // Time
      r.auditor     || "",   // Auditor
      r.designation || "",   // Designation
      r.division    || "",   // Division
      r.qno         || "",   // Q.No
      r.question    || "",   // Question
      r.audit       || "",   // Audit
      r.remarks     || "",   // Remarks
      driveUrl               // Attachment — Drive URL (blank if no photo)
    ]);
  });

  sheet.autoResizeColumns(1, MASTER_HEADERS.length);
  return { status: "success" };
}

    var hr = sheet.getRange(1, 1, 1, MASTER_HEADERS.length);
    hr.setBackground("#006b3f");
    hr.setFontColor("#ffffff");
    hr.setFontWeight("bold");
    hr.setHorizontalAlignment("center");
    sheet.setFrozenRows(1);
  }
  SpreadsheetApp.getActiveSpreadsheet().toast("Master sheet ready!", "Setup Done", 5);
}

// ── Login verify — reads Users sheet (A=Name, B=Password) ──
function verifyLogin(username, password) {
  if (!username || !password)
    return { status: "error", message: "Please enter username and password." };
  var ss    = SpreadsheetApp.getActiveSpreadsheet();
  var sheet = ss.getSheetByName("Users");
  if (!sheet) return { status: "error", message: "Users sheet not found." };
  var data = sheet.getDataRange().getValues();
  for (var i = 1; i < data.length; i++) {
    var name = String(data[i][0] || "").trim();
    var pass = String(data[i][1] || "").trim();
    if (name.toLowerCase() === username.toLowerCase() && String(pass) === String(password))
      return { status: "success", name: name };
  }
  return { status: "error", message: "Invalid username or password." };
}
